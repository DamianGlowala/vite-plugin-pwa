import type { Component } from 'solid-js'
import { Show } from 'solid-js'
import styles from  './PWAPrompt.module.css'

import { useRegisterSW } from 'virtual:pwa-register/solid'

function PWAPrompt() {
  const {
    offlineReady: [offlineReady, setOfflineReady],
  } = useRegisterSW({
    onRegistered(r) {
      r && setInterval(() => {
        // eslint-disable-next-line no-console
        console.log('Checking for sw update')
        r.update()
      }, 20000 /* 20s for testing purposes */)
    },
    onRegisterError(error) {
      // eslint-disable-next-line no-console
      console.log('SW registration error', error)
    },
  })

  const close = () => {
    setOfflineReady(false)
  }

  return (
    <div class={styles.PWAPromptContainer}>
      <Show when={offlineReady()}>
        <div class={styles.PWAPromptToast}>
          <div class={styles.PWAPromptMessage}>
            <span>App ready to work offline.</span>
          </div>
          <button class={styles.PWAPromptToastButton} onClick={() => close()}>Close</button>
        </div>
      </Show>
    </div>
  )
}

export default PWAPrompt
